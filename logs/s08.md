 - def count_vowels(word: str) -> dict:
    vowels = set("aeiou")
    word = word.lower()
    found = [ch for ch in word if ch in vowels]
    return {
        "word": word,
        "total": len(found),
        "breakdown": {v: found.count(v) for v in sorted(set(found))}
    }
 - if __name__ == "__main__":
    result = count_vowels("banana")
    print(f"Word     : {result['word']}")
    print(f"Vowels   : {result['total']}")
    print(f"Breakdown: {result['breakdown']}")
